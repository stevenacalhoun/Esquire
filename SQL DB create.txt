CREATE TABLE groups
(
groupID CHAR(40),
name CHAR(40),
admin CHAR(40),
description CHAR(40),
PRIMARY KEY (groupID)
);

CREATE TABLE users
(
email CHAR(40),
firstName CHAR(40),
lastName CHAR(40),
password CHAR(40),
phoneNum CHAR(40),
phoneEmail CHAR(40),
textUpdates CHAR(40),
emailUpdates CHAR(40),
PRIMARY KEY (email)
);

CREATE TABLE member_of
(
email CHAR(40),
groupID CHAR(40),
joinDate CHAR(40),
PRIMARY KEY (email,groupID)
);

ALTER TABLE member_of ADD FOREIGN KEY email_idxfk (email) REFERENCES users (email);

ALTER TABLE member_of ADD FOREIGN KEY groupID_idxfk (groupID) REFERENCES groups (groupID);
